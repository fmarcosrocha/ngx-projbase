Radio Button

O componente Radio permite selecionar apenas uma opção em uma lista de opções.

{{ elemento }} 
<mat-radio-group color="primary" class="mat-radio-column" [(ngModel)]="item">
  <mat-radio-button class="example-radio-button" *ngFor="let elemento of itens" [value]="elemento">
    {{ '{{' }} elemento {{ '}}' }}
  </mat-radio-button>
</mat-radio-group>

item?: string;
itens: string[] = ['Angular', 'React', 'Vuejs'];
