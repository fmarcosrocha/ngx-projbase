Input

It is the component that allows the input of textual data by the user.

Label cancel Value is required 
<form class="submit-form">
  <mat-label>Label</mat-label>
  <mat-form-field appearance="outline" class="w-100 mat-form-field-label-out" floatLabel="never">
    <input matInput type="text" placeholder="input example" name="value" [(ngModel)]="value"
      [formControl]="inputFormControl">

    <button *ngIf="value" matSuffix mat-icon-button aria-label="Clear" (click)="value = ''">
      <mat-icon>close</mat-icon>
    </button>

    <mat-error *ngIf="inputFormControl.hasError('required')">
      <mat-icon aria-hidden="false">cancel</mat-icon>
      Value is <strong>required</strong>
    </mat-error>
  </mat-form-field>
</form>

inputFormControl = new FormControl('', [
  Validators.required
]);

public value = '';
