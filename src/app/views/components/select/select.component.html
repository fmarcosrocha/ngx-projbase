Select

The select component lets you select items from a list of options.

Choice an option Option 1 Option 2 (disabled) Option 3 cancel Value ie required 
<form class="submit-form">
  <mat-label>Choice an option</mat-label>
  <mat-form-field appearance="outline" class="w-100 mat-form-field-label-out" floatLabel="never">
    <mat-select name="value" [(ngModel)]="value" [formControl]="inputFormControl">
      <mat-option value="option1">Option 1</mat-option>
      <mat-option value="option2" disabled>Option 2 (disabled)</mat-option>
      <mat-option value="option3">Option 3</mat-option>
    </mat-select>

    <mat-error *ngIf="inputFormControl.hasError('required')">
      <mat-icon aria-hidden="false">cancel</mat-icon>
      Value ie <strong>required</strong>
    </mat-error>
  </mat-form-field>
</form>

inputFormControl = new FormControl('', [
  Validators.required
]);

public value = '';
