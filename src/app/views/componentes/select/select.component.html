Select

O componente select permite selecionar itens em uma lista de opções.

Escolha uma opção Option 1 Option 2 (disabled) Option 3 cancel Valor é obrigatório 
<form class="submit-form">
  <mat-label>Escolha uma opção</mat-label>
  <mat-form-field appearance="outline" class="w-100 mat-form-field-label-out" floatLabel="never" color="accent">
    <mat-select name="valor" [(ngModel)]="valor" [formControl]="inputFormControl">
      <mat-option value="option1">Option 1</mat-option>
      <mat-option value="option2" disabled>Option 2 (disabled)</mat-option>
      <mat-option value="option3">Option 3</mat-option>
    </mat-select>

    <mat-error *ngIf="inputFormControl.hasError('required')">
      <mat-icon aria-hidden="false">cancel</mat-icon>
      Valor é <strong>obrigatório</strong>
    </mat-error>
  </mat-form-field>
</form>

inputFormControl = new FormControl('', [
  Validators.required
]);

public valor = '';
