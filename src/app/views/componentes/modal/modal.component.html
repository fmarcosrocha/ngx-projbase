Modal

Modal é uma janela que exibe um conteúdo adicional em uma camada acima da página atual, com uma sobreposição de superfície (overlay) cobrindo a página e tornando-a temporáriamente inacessível.

 
<button mat-stroked-button color="primary" class="rounded" (click)="abrirModal()">
  <div class="d-flex">
    <mat-icon class="mr-3">open_in_new</mat-icon>
    <span>Abrir modal</span>
  </div>
</button>

export class ModalComponent {{ '{' }}
  constructor(public _dialog: MatDialog) {{ '{ }' }}

  abrirModal(): void {{ '{' }}
    const dialogRef = this._dialog.open(ModalExemploComponent, {{ '{' }}
      width: '450px'
    {{ '}' }})
  {{ '}' }}
{{ '}' }}

@Component({{ '{' }}
  selector: 'app-modal-exemplo',
  templateUrl: 'modal-exemplo.component.html',
{{ '}' }})
export class ModalExemploComponent {{ '{ }' }}

  constructor(public dialogRef: MatDialogRef<ModalExemploComponent>) {{ '{' }}

  onNoClick(): void {{ '{' }}
    this.dialogRef.close();
    {{ '}' }}
{{ '}' }}
