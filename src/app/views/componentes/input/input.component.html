Input

É o componente que permite a entrada de dados textuais por parte do usuário.

Exemplo de label cancel Valor é obrigatório 
<form class="submit-form">
  <mat-label>Exemplo de label</mat-label>
  <mat-form-field appearance="outline" class="w-100 mat-form-field-label-out" floatLabel="never" color="accent">
    <input matInput type="text" placeholder="Exemplo de input" name="valor" [(ngModel)]="valor"
      [formControl]="inputFormControl">

    <button *ngIf="valor" matSuffix mat-icon-button aria-label="Clear" (click)="valor = ''">
      <mat-icon>close</mat-icon>
    </button>

    <mat-error *ngIf="inputFormControl.hasError('required')">
      <mat-icon aria-hidden="false">cancel</mat-icon>
      Valor é <strong>obrigatório</strong>
    </mat-error>
  </mat-form-field>
</form>

inputFormControl = new FormControl('', [
  Validators.required
]);

public valor = '';
